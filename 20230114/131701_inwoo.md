#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

// number_len은 배열 number의 길이입니다.
int solution(int number[], size_t number_len) {
    int answer = 0;
    int num1, num2, num3 = 0;
    for(int i= 0; i<number_len-2; i++){
        num1 = number[i];
        for(int j = i+1; j<number_len-1; j++){
            num2 = number[j];
                for(int k = j+1; k<number_len; k++){
                    num3 = number[k];
                    if(num1+num2+num3 == 0){
                        answer++;
                    }
                    else
                        continue;
                }
        }
    }
    return answer;
}
